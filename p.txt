import json

class Company:
    def __init__(self, name, location, est_year, employee_count, ceo, description):
        self.name = name
        self.location = location
        self.est_year = est_year
        self.employee_count = employee_count
        self.ceo = ceo
        self.description = description

    def to_dict(self):
        return {
            "name": self.name,
            "location": self.location,
            "established_year": self.est_year,
            "employee_count": self.employee_count,
            "ceo": self.ceo,
            "description": self.description
        }

def store_in_json(company_obj, json_file='companies.json'):
    company_dict = company_obj.to_dict()
    
    try:
        # Read existing data
        with open(json_file, 'r') as file:
            data = json.load(file)
    except FileNotFoundError:
        data = []

    # Append new data
    data.append(company_dict)

    # Write back to file
    with open(json_file, 'w') as file:
        json.dump(data, file, indent=4)
